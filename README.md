# Instalar o nodejs

npm install nodejs
node -v

# Instalar o npm

curl -L https://npmjs.org/install.sh | sudo sh

# Criar um projeto

npm init -y

# Instalar o typescript

npm i typescript -D
npm i @types/node -D

# configurar o typescript no projeto

npx tsc --init

# Instalar o tsx converte automaticamente ts para js.
npm i tsx -D

# Instalar o framework fastify

npm i fastify
